<div className="grid gap-3">

                {(staffDirectory || []).map((s) => {

                  const impact = staffImpact.find((i) => i.faculty_id === s.id || i.faculty_name.toLowerCase() === (s.name || '').toLowerCase());

                  const achievements = impact

                    ? [

                        `Leads ${impact.subjects.length} subject(s)`,

                        impact.subjects.some((sub) => sub.failure_rate < 20) ? 'Low failure rates' : 'Stabilizing subjects',

                        impact.subjects.some((sub) => sub.average_marks >= 80) ? 'High average marks' : 'Improving averages',

                      ]

                    : ['Awaiting timetable data'];

                  return (

                    <div key={s.id} className="row-card flex flex-col gap-3">

                      <div className="flex items-start justify-between gap-3">

                        <div className="flex items-center gap-3">

                          <div className="h-10 w-10 rounded-full bg-primary/10 text-primary flex items-center justify-center font-bold">

                            {s.name?.slice(0, 1) || s.username.slice(0, 1)}

                          </div>

                          <div>

                            <p className="text-sm font-semibold text-foreground">{s.name}</p>

                            <p className="text-xs text-muted-foreground">{s.username} � {s.department || 'Dept NA'}</p>

                            <p className="text-[11px] text-muted-foreground">{s.email || 'No email'}</p>

                          </div>

                        </div>

                        <div className="flex items-center gap-2">

                          <button

                            className="tab-chip"

                            onClick={() => {

                              setEditingStaff(s);

                              setStaffForm({

                                username: s.username,

                                name: s.name,

                                email: s.email || '',

                                department: s.department || '',

                                password: '',

                              });

                            }}

                          >

                            Edit

                          </button>

                          <button

                            className="tab-chip !bg-rose-500/10 !text-rose-600 hover:!bg-rose-500/20"

                            onClick={() => {

                              if (confirm(`Delete staff ${s.name || s.username}?`)) {

                                deleteStaffMutation.mutate(s.id);

                              }

                            }}

                            disabled={deleteStaffMutation.isPending}

                          >

                            Delete

                          </button>

                        </div>

                      </div>

                      {impact && (

                        <div className="flex flex-wrap gap-2">

                          {impact.subjects.map((sub) => (

                            <span key={sub.subject_code} className="px-2 py-1 rounded-full bg-muted text-[10px] font-black uppercase tracking-widest">

                              {sub.subject_code} � {sub.subject_name}

                            </span>

                          ))}

                        </div>

                      )}

                      <div className="text-[11px] text-muted-foreground">

                        Achievements: {achievements.join(' | ')}

                      </div>

                    </div>

                  );

                })}

                {loadingStaff && <p className="text-sm text-muted-foreground">Loading staff...</p>}

                {!loadingStaff && (staffDirectory?.length ?? 0) === 0 && (

                  <p className="text-sm text-muted-foreground">No staff added yet.</p>

                )}

              </div>

            </div>

          </article>

        </div>

      )}



      {/* Profile Modal */}

      <StudentProfile360 rollNo={selectedRollNo} onClose={handleCloseProfile} />

    </div>

  );

}
