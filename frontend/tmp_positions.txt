                  const achievements = impact

                    ? [

                        `Leads ${impact.subjects.length} subject(s)`,

                        impact.subjects.some((sub) => sub.failure_rate < 20) ? 'Low failure rates' : 'Stabilizing subjects',

                        impact.subjects.some((sub) => sub.average_marks >= 80) ? 'High average marks' : 'Improving averages',

                      ]

                  return (

                          </div>

                          <div>

                              });

                          <button

                            onClick={() => {

                              if (confirm(`Delete staff ${s.name || s.username}?`)) {

                                deleteStaffMutation.mutate(s.id);

                              }

                            }}

                            disabled={deleteStaffMutation.isPending}

                          >

                      {impact && (

                          {impact.subjects.map((sub) => (

                              {sub.subject_code} � {sub.subject_name}
